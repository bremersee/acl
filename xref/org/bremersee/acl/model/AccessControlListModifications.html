17  package org.bremersee.acl.model;
18  
19  import com.fasterxml.jackson.annotation.JsonIgnore;
20  import com.fasterxml.jackson.databind.annotation.JsonDeserialize;
21  import io.swagger.v3.oas.annotations.media.Schema;
22  import java.util.Collection;
23  import java.util.List;
24  import java.util.stream.Collectors;
25  import java.util.stream.Stream;
26  import org.immutables.value.Value;
27  
28  /**
29   * Specifies modifications of an access control list.
30   *
31   * @author Christian Bremer
32   */
33  @Value.Immutable
34  @Schema(description = "Specifies modifications of an access control list.")
35  @JsonDeserialize(builder = ImmutableAccessControlListModifications.Builder.class)
36  public interface AccessControlListModifications {
37  
38    /**
39     * Creates new builder.
40     *
41     * @return the access control list modifications builder
42     */
43    static ImmutableAccessControlListModifications.Builder builder() {
44      return ImmutableAccessControlListModifications.builder();
45    }
46  
47    /**
48     * Get modifications.
49     *
50     * @return the modifications
51     */
52    @Schema(description = "The access control entry modifications.")
53    @Value.Default
54    default Collection<AccessControlEntryModifications> getModifications() {
55      return List.of();
56    }
57  
58    /**
59     * Gets modifications distinct.
60     *
61     * @return the modifications distinct
62     */
63    @Schema(hidden = true)
64    @JsonIgnore
65    @Value.Derived
66    default Collection<AccessControlEntryModifications> getModificationsDistinct() {
67      return getModifications()
68          .stream()
69          .collect(Collectors.toMap(
70              AccessControlEntryModifications::getPermission,
71              mod -> mod,
72              (first, second) -> AccessControlEntryModifications.builder()
73                  .from(first)
74                  .isGuest(first.isGuest() && second.isGuest())
75                  .addUsers(Stream
76                      .concat(first.getAddUsers().stream(), second.getAddUsers().stream())
77                      .collect(Collectors.toSet()))
78                  .removeUsers(Stream
79                      .concat(first.getRemoveUsers().stream(), second.getRemoveUsers().stream())
80                      .collect(Collectors.toSet()))
81                  .addRoles(Stream
82                      .concat(first.getAddRoles().stream(), second.getAddRoles().stream())
83                      .collect(Collectors.toSet()))
84                  .removeRoles(Stream
85                      .concat(first.getRemoveRoles().stream(), second.getRemoveRoles().stream())
86                      .collect(Collectors.toSet()))
87                  .addGroups(Stream
88                      .concat(first.getAddGroups().stream(), second.getAddGroups().stream())
89                      .collect(Collectors.toSet()))
90                  .removeGroups(Stream
91                      .concat(first.getRemoveGroups().stream(), second.getRemoveGroups().stream())
92                      .collect(Collectors.toSet()))
93                  .build()))
94          .values();
95    }
96  
97  }
